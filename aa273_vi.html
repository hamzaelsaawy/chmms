T = 300

for i = 1:T
## E Step
    map!(digamma, log_a, s)
    log_a .-= digamma(sum(s, 2))
    #map!(exp, a)

    # - E[(x_n - _j)^T _j (...)]
    for j in 1:K, n in 1:N
        t = X[n, :] - ms[j, :]
        log_b[n, j, :] = t' * Ws[j] * t
    end
    log_b .*= - vs'
    log_b .-= D ./ s'

    # + E[ln _j]
    log_b .+= sum(digamma((vs + 1 - d) / 2) for d in 1:D)'
    log_b .+= log(map(det, Ws))'
    log_b .+= D*log(2)

    log_b -= D * log(2*)
    log_b /= 2
    #map!(exp, log_b)

    # Baum Welch pass
    log_v[1, :] = log(0) .+ log_b[1, :]
    temp = similar(log_a)
    for n = 2:N
        temp[:] = log_v[n-1, :] .+ log_a
        temp_max = maximum(temp)
        log_v[n, :] = log_b[n, :] .+ log( sum(exp(temp - temp_max), 1) )'
        log_v[n, :] += temp_max
    end

    log_w[N, :] = 0
    for n = (N-1):-1:1
        temp[:] = log_a .+ log_b[n+1, :]' .+ log_w[n+1, :]'
        temp_max = maximum(temp)
        log_w[n, :] = log( sum(exp(temp - temp_max), 2) ) + temp_max
    end
    
     = log_v .+ log_w
    for n = 1:N
        temp_max = maximum([n, :])
        [n, :] -= log(sum(exp([n, :] - temp_max))) + temp_max
    end
    map!(exp, )
    0 = [1, :]
    
    for n = 2:N
        [n, :, :] = log_v[n-1, :] .+ log_a .+ log_b[n, :]' .+ log_w[n, :]'
        temp_max = maximum([n, :, :])
        [n, :, :] -= log(sum(exp([n, :, :] - temp_max))) + temp_max
    end
    map!(exp, )
    [1, :, :] = 0

    ## M Step
    Ns[:] = sum(, 1)

    fill!(x_bars, 0)
    for j in 1:K, n in 1:N
        x_bars[j, :] += [n, j] * X[n, :]
    end
    x_bars ./= Ns

    for j in 1:K 
        fill!(Ss[j], 0)
        for n in 1:N
            t = X[n, :] - x_bars[j, :]
            Ss[j] += [n, j] * t * t'
        end
        Ss[j] ./= Ns[j]
    end

    s = 0 + Ns
    vs = v0 + Ns
    ms = ((0 * m0)' .+ x_bars .* Ns) ./ s

    for j = 1:K
        Nj = Ns[j]
        t = x_bars[j, :] - m0
        Ws[j] = inv(invW0 + Nj*Ss[j] + 0*Nj/(0 + Nj) * t * t')
    end

    # transition
    s = s0 + squeeze(sum(, 1), 1)

    # starting
     = 0 + Ns
end
